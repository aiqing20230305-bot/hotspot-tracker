### 3.2 跨平台话题合并算法

**问题**：当前仅按标题前10字匹配，准确率低

**改进方案**：

```python
def merge_cross_platform_topics(all_items):
    """智能合并跨平台话题"""
    from difflib import SequenceMatcher
    
    topics = []
    used = set()
    
    for item in all_items:
        if id(item) in used:
            continue
        
        # 找相似话题
        similar = [item]
        for other in all_items:
            if id(other) in used or other['platform'] == item['platform']:
                continue
            
            # 计算标题相似度
            ratio = SequenceMatcher(None, item['title'], other['title']).ratio()
            if ratio > 0.7:  # 相似度阈值
                similar.append(other)
                used.add(id(other))
        
        used.add(id(item))
        
        if len(similar) > 1:
            platforms = [s['platform'] for s in similar]
            total_hot = sum(s.get('hot_value', 0) for s in similar)
            topics.append({
                'title': max(similar, key=lambda x: x.get('hot_value', 0))['title'],
                'platforms': platforms,
                'total_hot': total_hot,
                'cross_platform': True,
                'items': similar
            })
    
    return sorted(topics, key=lambda x: x['total_hot'], reverse=True)
```

### 3.3 创意建议差异化

**问题**：所有行业建议相同，无针对性

**改进方案**：

```python
def generate_smart_recommendations(industry, hotspots, trends):
    """基于热点生成差异化建议"""
    
    ideas = []
    
    # 1. 基于热点类型
    if hotspots:
        top = hotspots[0]
        title = top['title']
        
        # 判断热点类型
        if any(kw in title for kw in ['爆料', '泄露', '翻车', '道歉']):
            ideas.append(f"【危机公关】围绕「{title[:10]}」，做品牌避坑/行业反思内容")
        elif any(kw in title for kw in ['测评', '开箱', '推荐']):
            ideas.append(f"【跟风测评】做同类产品横向对比，蹭{title[:10]}热度")
        elif any(kw in title for kw in ['穿搭', '造型', '风格']):
            ideas.append(f"【场景种草】拍摄「{title[:10]}」真人上身/使用场景")
        else:
            ideas.append(f"【热点解读】快速输出行业视角解读「{title[:10]}」")
    
    # 2. 基于趋势标签
    if '挑战' in trends:
        ideas.append("【参与挑战】发起或参与相关话题挑战，提升曝光")
    if '开箱' in trends or '测评' in trends:
        ideas.append("【产品测评】制作开箱/测评视频，突出产品卖点")
    
    # 3. 基于行业特性
    industry_specific = {
        "美妆": "【教程种草】拍摄上妆教程，突出产品使用效果",
        "母婴": "【育儿干货】输出实用育儿知识，建立专业人设",
        "数码": "【参数解读】用通俗语言解读技术参数，降低认知门槛",
        "服装": "【穿搭示范】真人穿搭展示，提供搭配灵感",
        "食品": "【口味测评】真实试吃体验，强调口味/口感",
        "汽车": "【试驾体验】实车试驾分享，展示驾驶感受",
        "大健康": "【科普辟谣】用专业视角解读健康热点，建立信任",
        "快消": "【对比测评】同类产品横向对比，突出性价比",
        "家电": "【使用场景】展示产品解决生活痛点的实际场景"
    }
    
    if industry in industry_specific:
        ideas.append(industry_specific[industry])
    
    return ideas[:3]  # 最多3条建议
```

## 四、输出格式选项
